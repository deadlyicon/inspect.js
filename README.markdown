# Inspect

    var names = ['frank', 'sam', 'steve'];

    names.inspect();
    //-> ['frank', 'sam', 'steve']

    Object.inspect(names);
    //-> ['frank', 'sam', 'steve']

    var colors = {red:'335522', blue:'1234'};

    colors.inspect
    //-> undefined

    Object.inspect(colors);
    //-> {red:'335522', blue:'1234'}
